 import Qt.labs.shaders 1.0

 Item {
     anchors.fill: parent

     Image {
         id: image
         width: parent.width
         height: parent.height * 0.65
         source: "images/sky.jpg"
         smooth: true
     }
     WaterEffect {
         sourceItem: image
         intensity: 5
         height: parent.height - image.height
