 import QtQuick 1.0
 import Qt.labs.shaders 1.0

 Item {
     id: main
     anchors.fill: parent

     ShaderEffectSource {
         id: thesource
         sourceItem: Image { source: "images/wave.jpg" }
         live: false
         hideSource: true
     }

     RadialWaveEffect {
         id: layer
         anchors.fill: parent;
         source: thesource

         wave: 0.0
         waveOriginX: 0.5
         waveOriginY: 0.5
         waveWidth: 0.01

         NumberAnimation on wave {
             id: waveAnim
             running: true
             loops: Animation.Infinite
             easing.type: "Linear"
             from: 0.0000; to: 2.0000;
             duration: 2500
         }
     }

     MouseArea {
         anchors.fill: parent
         onPressed: {
             waveAnim.stop()
             layer.waveOriginX = mouseX / main.width
             layer.waveOriginY = mouseY / main.height
             waveAnim.start()
         }
     }
 }
